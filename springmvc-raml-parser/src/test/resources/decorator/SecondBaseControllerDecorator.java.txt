package com.gen.test;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.*;
import java.util.*;
import org.springframework.web.bind.annotation.*;
import com.gen.test.model.*;


@RestController
@RequestMapping("/api/secondBase", produces="application/json")
public class SecondBaseControllerDecorator implements SecondBaseController {

    @Autowired
    private SecondBaseController secondBaseControllerDelegate;

	/**
	 * No description
	 */
	@RequestMapping(value="", method=RequestMethod.GET)
	public ResponseEntity getSecondBase () {
        return this.secondBaseControllerDelegate.getSecondBase();
	}

	/**
	 * No description
	 */
	@RequestMapping(value="/endpointWithGet", method=RequestMethod.GET)
	public ResponseEntity getEndpointWithGet () {
        return this.secondBaseControllerDelegate.getEndpointWithGet();
    }

	/**
	 * No description
	 */
	@RequestMapping(value="/endpointWithURIParam/{uriParam}", method=RequestMethod.GET)
	public ResponseEntity getEndpointWithURIParamByUriParam (@PathVariable String uriParam) {
        return this.secondBaseControllerDelegate.getEndpointWithURIParamByUriParam(uriParam);
	}

}
